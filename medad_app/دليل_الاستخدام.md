## 1️⃣ نظام التحقق من الإدخال (Validators)

### الاستخدام الأساسي

#### في نموذج (Form):

```dart
import 'package:medad_app/utils/validators.dart';

class MyFormScreen extends StatefulWidget {
  @override
  _MyFormScreenState createState() => _MyFormScreenState();
}

class _MyFormScreenState extends State<MyFormScreen> {
  final _formKey = GlobalKey<FormState>();
  final _phoneController = TextEditingController();
  final _passwordController = TextEditingController();

  void _submit() {
    // التحقق من جميع الحقول
    if (_formKey.currentState!.validate()) {
      // جميع الحقول صحيحة - المتابعة
      print('Form is valid!');
    }
  }

  @override
  Widget build(BuildContext context) {
    return Form(
      key: _formKey,
      child: Column(
        children: [
          // حقل رقم الجوال
          TextFormField(
            controller: _phoneController,
            keyboardType: TextInputType.phone,
            validator: Validators.validatePhone,
            decoration: InputDecoration(
              labelText: 'رقم الجوال',
            ),
          ),
          
          // حقل كلمة المرور
          TextFormField(
            controller: _passwordController,
            obscureText: true,
            validator: Validators.validatePassword,
            decoration: InputDecoration(
              labelText: 'كلمة المرور',
            ),
          ),
          
          ElevatedButton(
            onPressed: _submit,
            child: Text('إرسال'),
          ),
        ],
      ),
    );
  }
}
```

## 3️⃣ متغيرات البيئة (Environment Variables)

### إنشاء ملف .env جديد

1. **انسخ القالب**:
```bash
cp .env.example .env.local
```

2. **عدّل القيم**:
```env
FIREBASE_API_KEY=your_actual_api_key
FIREBASE_APP_ID_ANDROID=your_app_id
# ... باقي القيم
```

3. **التطبيق سيقرأها تلقائياً** من `main.dart`

### كيفية القراءة في الكود

```dart
import 'package:flutter_dotenv/flutter_dotenv.dart';

void readEnvVariables() {
  // قراءة متغير
  String apiKey = dotenv.env['FIREBASE_API_KEY'] ?? '';
  String projectId = dotenv.env['FIREBASE_PROJECT_ID'] ?? '';
  
  // التحقق من التحميل
  if (dotenv.isLoaded) {
    print('Environment loaded!');
  }
  
  print('API Key: $apiKey');
}
```

### بيئات مختلفة

#### للتطوير (Development):
```bash
# يتم تحميله تلقائياً من main.dart
await dotenv.load(fileName: '.env.development');
```

#### للإنتاج (Production):
```dart
// في main.dart
await dotenv.load(
  fileName: kReleaseMode ? '.env.production' : '.env.development',
);
```

#### محلي (Local - غير موجود في Git):
```bash
# أنشئ ملف محلي
cp .env.example .env.local

# عدّل القيم للاستخدام المحلي
# هذا الملف لن يتم رفعه لـ Git
```

## 4️⃣ ضغط الصور (Image Compression)

### الاستخدام الأساسي

```dart
import 'package:flutter_image_compress/flutter_image_compress.dart';
import 'package:path_provider/path_provider.dart';
import 'package:medad_app/utils/constants.dart';

Future<File?> compressImage(File imageFile) async {
  try {
    final dir = await getTemporaryDirectory();
    final targetPath = '${dir.path}/compressed_${DateTime.now().millisecondsSinceEpoch}.jpg';
    
    final result = await FlutterImageCompress.compressAndGetFile(
      imageFile.absolute.path,
      targetPath,
      quality: AppConstants.imageQuality, // 70
      minWidth: AppConstants.maxImageWidth, // 1024
      minHeight: AppConstants.maxImageWidth, // 1024
    );
    
    return result;
  } catch (e) {
    print('Error compressing image: $e');
    return null;
  }
}
```

### في AdService (مثال كامل)

```dart
import 'dart:io';
import 'package:firebase_storage/firebase_storage.dart';
import 'package:flutter_image_compress/flutter_image_compress.dart';
import 'package:path_provider/path_provider.dart';
import 'package:medad_app/utils/constants.dart';

class AdService {
  final FirebaseStorage _storage = FirebaseStorage.instance;

  /// رفع صور مضغوطة
  Future<List<String>> uploadImages(List<File> images, String adId) async {
    List<String> urls = [];
    
    for (int i = 0; i < images.length; i++) {
      // ضغط الصورة
      final compressedImage = await _compressImage(images[i]);
      if (compressedImage == null) continue;
      
      // رفع الصورة
      String fileName = 'ads/$adId/${DateTime.now().millisecondsSinceEpoch + i}_$i.jpg';
      Reference ref = _storage.ref().child(fileName);
      
      final bytes = await compressedImage.readAsBytes();
      final metadata = SettableMetadata(contentType: 'image/jpeg');
      final uploadTask = ref.putData(bytes, metadata);
      final snapshot = await uploadTask;
      final url = await snapshot.ref.getDownloadURL();
      
      urls.add(url);
    }
    
    return urls;
  }
  
  /// ضغط صورة
  Future<File?> _compressImage(File imageFile) async {
    try {
      final dir = await getTemporaryDirectory();
      final targetPath = '${dir.path}/compressed_${DateTime.now().millisecondsSinceEpoch}.jpg';
      
      final result = await FlutterImageCompress.compressAndGetFile(
        imageFile.absolute.path,
        targetPath,
        quality: AppConstants.imageQuality,
        minWidth: AppConstants.maxImageWidth,
        minHeight: AppConstants.maxImageWidth,
      );
      
      return result;
    } catch (e) {
      print('Error compressing image: $e');
      return imageFile; // إرجاع الصورة الأصلية عند الخطأ
    }
  }
}
```

### التحقق من حجم الصورة

```dart
import 'dart:io';

bool isImageSizeValid(File imageFile) {
  try {
    int sizeInBytes = imageFile.lengthSync();
    int sizeInMB = sizeInBytes ~/ (1024 * 1024);
    
    return sizeInMB <= AppConstants.maxImageSizeMB;
  } catch (e) {
    return false;
  }
}

// الاستخدام
void uploadCheck() {

  if (!isImageSizeValid(image)) {
    print('الصورة كبيرة جداً. الحد الأقصى: ${AppConstants.maxImageSizeMB}MB');
    return;
  }

}
```

```dart
  Future<void> loadAds() async {
    if (_isLoading) return;

    try {
      // مثال: تحميل من Firestore
      final snapshot = await FirebaseFirestore.instance
          .collection('ads')
          .where('isApproved', isEqualTo: true)
          .orderBy('createdAt', descending: true)
          .limit(AppConstants.adsPageSize) // 20 إعلان
          .get();
      
      setState(() {
        _ads = snapshot.docs
            .map((doc) => AdModel.fromJson(doc.data(), doc.id))
            .toList();
        _hasMore = _ads.length == AppConstants.adsPageSize;
      });

    } finally {
      setState(() => _isLoading = false);
```

### الاستخدام الأساسي

```dart
import 'package:logger/logger.dart';
import 'package:medad_app/utils/constants.dart';

class AppLogger {
  static final Logger _logger = Logger(
    printer: PrettyPrinter(
      methodCount: 2,
      errorMethodCount: 8,
      lineLength: 120,
      colors: true,
      printEmojis: true,
    ),
  );
  
  static void info(String message) {
    _logger.i('ℹ️ $message');
  }
  
  static void warning(String message) {
    _logger.w('⚠️ $message');
  }
  
  static void error(String message, [dynamic err, StackTrace? stackTrace]) {
    _logger.e('❌ $message', error: err, stackTrace: stackTrace);
  }
  
  static void debug(String message) {
    if (kDebugMode) {
      _logger.d('🔍 $message');
    }
  }
}

// الاستخدام
void example() {
  AppLogger.info('User logged in successfully');
  AppLogger.warning('Low storage space');
  AppLogger.error('Failed to upload image', exception, stackTrace);
  AppLogger.debug('Loading ads with page size: ${AppConstants.adsPageSize}');
}
```

## 7️⃣ Network Connectivity

### التحقق من الاتصال

```dart
import 'package:connectivity_plus/connectivity_plus.dart';

class NetworkHelper {
  static Future<bool> isOnline() async {
    final connectivityResult = await Connectivity().checkConnectivity();
    return connectivityResult != ConnectivityResult.none;
  }
  
  static Future<String> getConnectionType() async {
    final connectivityResult = await Connectivity().checkConnectivity();
    
    switch (connectivityResult) {
      case ConnectivityResult.wifi:
        return 'WiFi';
      case ConnectivityResult.mobile:
        return 'Mobile';
      case ConnectivityResult.none:
        return 'No Connection';
      default:
        return 'Unknown';
    }
  }
}

// الاستخدام
void checkConnection() async {
  if (await NetworkHelper.isOnline()) {
    print('Online! Connection: ${await NetworkHelper.getConnectionType()}');
    // تحميل بيانات من الإنترنت
  } else {
    print('Offline - Using cached data');
    // استخدام البيانات المخزنة محلياً
  }
}
```

### الاستماع لتغيرات الاتصال

```dart
import 'package:connectivity_plus/connectivity_plus.dart';

class NetworkMonitor {
  StreamSubscription? _subscription;
  
  void startListening() {
    _subscription = Connectivity()
        .onConnectivityChanged
        .listen((ConnectivityResult result) {
      
      if (result == ConnectivityResult.none) {
        print('Lost connection');
        // عرض رسالة Offline
      } else {
        print('Connected via: $result');
        // إعادة تحميل البيانات
      }
    });
  }
  
  void dispose() {
    _subscription?.cancel();
  }
}
```

```dart
import 'package:flutter/material.dart';
import 'package:medad_app/utils/validators.dart';
import 'package:medad_app/utils/constants.dart';

class AddAdScreen extends StatefulWidget {
  @override
  _AddAdScreenState createState() => _AddAdScreenState();
}

class _AddAdScreenState extends State<AddAdScreen> {
  final _formKey = GlobalKey<FormState>();
  final _titleController = TextEditingController();
  final _descriptionController = TextEditingController();
  final _priceController = TextEditingController();
  final _locationController = TextEditingController();
  bool _isLoading = false;

  Future<void> _submit() async {
    // التحقق من جميع الحقول
    if (!_formKey.currentState!.validate()) {
      return;
    }

    setState(() => _isLoading = true);

    try {
      // تنظيف الإدخال
      String title = Validators.sanitizeInput(_titleController.text);
      String description = Validators.sanitizeInput(_descriptionController.text);
      
      // إنشاء الإعلان
      // ... كود إنشاء الإعلان
      
      ScaffoldMessenger.of(context).showSnackBar(
        SnackBar(
          content: Text('تم إنشاء الإعلان بنجاح'),

      ScaffoldMessenger.of(context).showSnackBar(

          content: Text('حدث خطأ: $e'),

              controller: _priceController,
```
